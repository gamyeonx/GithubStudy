namespace 코딩테스트용;

class Program
{
    static void Main(string[] args)
    {

    }
    public int solution(int[] array, int height) 
    {
        int answer = 0;
        for (int i = 0; i < array.Length; i++)
        {
            if (array[i] > height)
            {
                answer = i;
            }
        }
            
            
        return answer;
    }

}
